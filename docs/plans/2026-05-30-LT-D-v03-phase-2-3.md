## 1. 目標(一句)

把 op-assistant 每天 09:00 的 daily_curate gemma4 整理結果,**從只寫 events.payload** 升級成**正式寫入 `improvement_candidates` row**(Phase 2);同時讓 Telegram 推給 Gary 的訊息**附 inline keyboard 按鈕**(`✅批准 / ❌拒絕 / 🔍看 / 💥KILL`),callback_data 帶 candidate_id(Phase 3)。

## 2. 背景(Codex 必讀)

1. **現況**:`~/.hermes/profiles/op-assistant/scripts/op_assistant_daily_curate.py` 每天 09:00 跑,gemma4 整理過去 24h OP traffic,寫一筆 `events.event_type='daily_curate_summary'`,payload 內含 actionable list,然後推 Telegram 純文字訊息給 Gary。
2. **缺口**:gemma 整理出的 actionable(可能新 keyword、新 regex、新 router rule)目前**只**存在 events.payload,沒有結構化、沒有可批准 ID、Gary 看了也按不下去。
3. **V0.3 Phase 2 解法**:每個 actionable 都建一個 `improvement_candidates` row,結構化、有 UUID candidate_id、有 typed_payload(JSON)、有 payload_hash(SHA256 of stable JSON repr)、有 proposed_by_agent_id 指向 `agents.name='op-assistant'`。
4. **V0.3 Phase 3 解法**:Telegram 訊息附 inline keyboard,每個 actionable 一組 4 個按鈕(批准/拒絕/看/kill),callback_data 編碼為 `apv:<short_id>` `rej:<short_id>` `vw:<short_id>` `kill:<short_id>`(<short_id> 是 candidate_id 前 8 hex char,訊息文末附短編號 ↔ candidate_id 對照表)。
5. **鐵則(不可破)**:`telegram-op-control/adapter.py` 第 1-30 行明寫「There is no LLM import in this file」— **本 LT 動的 dispatcher.py / 新增的 send 模組也都不准 import LLM**。Inline keyboard JSON 結構必須 hard-code 模板,**不能** LLM 生成。
6. **gpt-oss:120b swap** 等 LT-E + LT-G,本 LT 還是用 gemma4:e4b(daily_curate.py line 76 不動)。

### Step 1 — `improvement_candidates` 寫入 helper(新檔)

新檔:`~/.hermes/profiles/op-assistant/scripts/closed_loop_kernel/candidate_writer.py`

```python
"""Write improvement_candidates rows from daily_curate gemma output.

V0.3 Phase 2. NO LLM. Deterministic mapping from gemma's
structured `actionable` items into typed_payload rows.
"""
from __future__ import annotations

import hashlib
import json
import uuid
from typing import Any

from .store import KernelStore


def _payload_hash(typed_payload: dict) -> str:
    """Stable SHA256 of typed_payload — sort keys, no whitespace."""
    canonical = json.dumps(typed_payload, sort_keys=True, separators=(",", ":"))
    return hashlib.sha256(canonical.encode("utf-8")).hexdigest()


def write_candidates_from_gemma_summary(
    *,
    store: KernelStore,
    summary: dict,
    proposed_by_agent_name: str = "op-assistant",
    source_event_id: str,
) -> list[str]:
    """Iterate summary['actionable'] and INSERT one improvement_candidates row each.

    Returns list of candidate_id strings (UUIDs).
    """
    agent_id = store.fetch_one(
        "SELECT id FROM agents WHERE name = %s", (proposed_by_agent_name,)
    )["id"]

    created_ids: list[str] = []
    for item in summary.get("actionable", []):
        # item shape from gemma's typed prompt:
        #   { "type": "keyword"|"regex"|"intent_alias",
        #     "value": "...",
        #     "rationale": "...",
        #     "example_failure_ids": ["..."] }
        typed_payload = {
            "type": item["type"],
            "value": item["value"],
            "rationale": item.get("rationale", ""),
            "example_failure_ids": item.get("example_failure_ids", []),
        }
        candidate_id = str(uuid.uuid4())
        store.execute(
            """
            INSERT INTO improvement_candidates (
                id, proposed_by_agent_id, source_event_id,
                typed_payload, validation_assertions, rollback_plan,
                domain, created_at
            ) VALUES (
                %s, %s, %s, %s::jsonb, %s::jsonb, %s::jsonb, %s, now()
            )
            """,
            (
                candidate_id,
                agent_id,
                source_event_id,
                json.dumps(typed_payload),
                json.dumps({"sandbox_replay_must_pass": True}),  # filled by Phase 6
                json.dumps({"action": "revert_data_row", "details": "filled by Phase 7"}),
                proposed_by_agent_name,  # denormalized domain field
            ),
        )
        # Also store payload_hash in a column once Phase 5 adds it; for now skip
        created_ids.append(candidate_id)
    return created_ids
```

### Step 3 — Telegram inline keyboard send(新模組)

新檔:`~/.hermes/plugins/telegram-op-control/sender.py`

```python
"""Telegram outbound send for V0.3 Phase 3 — inline keyboard for candidates.

NO LLM. Pure template + deterministic candidate_id encoding.
"""
from __future__ import annotations

import json
import urllib.parse
import urllib.request
from typing import Any

TELEGRAM_API_BASE = "https://api.telegram.org/bot"

# 4 個按鈕的固定模板;絕不從 LLM 來
BUTTONS = ["✅ 批准", "❌ 拒絕", "🔍 看詳情", "💥 KILL"]
ACTION_PREFIX = ["apv", "rej", "vw", "kill"]


def _short(candidate_id: str) -> str:
    """前 8 個 hex char,讓 Gary 一眼看完。"""
    return candidate_id.replace("-", "")[:8]


def build_inline_keyboard(candidate_id: str) -> dict:
    return {
        "inline_keyboard": [[
            {"text": BUTTONS[i], "callback_data": f"{ACTION_PREFIX[i]}:{_short(candidate_id)}"}
            for i in range(4)
        ]]
    }


def send_daily_summary_with_candidates(
    *,
    bot_token: str,
    chat_id: int,
    summary_text: str,
    candidates: list[dict],   # [{ "candidate_id": "...", "type": "keyword", "value": "...", "rationale": "..." }, ...]
) -> list[int]:
    """推一條 header summary + 每個 candidate 一條訊息(各自帶 inline keyboard)。

    回 list[message_id] for audit.
    """
    msg_ids: list[int] = []

    # Header 訊息
    header = (
        f"📋 {summary_text}\n\n"
        f"今日 actionable 候選 {len(candidates)} 個,逐條按鈕回我:"
    )
    msg_ids.append(_send_text(bot_token, chat_id, header))

    # 每個 candidate
    for i, c in enumerate(candidates, 1):
        body = (
            f"#{i}  [{_short(c['candidate_id'])}]\n"
            f"類型: {c['type']}\n"
            f"提案: {c['value']}\n"
            f"理由: {c['rationale'][:200]}"
        )
        keyboard = build_inline_keyboard(c["candidate_id"])
        msg_ids.append(_send_text(bot_token, chat_id, body, reply_markup=keyboard))

    # 對照表(短編號 ↔ 完整 candidate_id),給 LT-E callback dispatcher 用
    mapping_lines = [f"{_short(c['candidate_id'])} → {c['candidate_id']}" for c in candidates]
    footer = "對照表:\n" + "\n".join(mapping_lines)
    msg_ids.append(_send_text(bot_token, chat_id, footer))

    return msg_ids


def _send_text(bot_token: str, chat_id: int, text: str, reply_markup: dict | None = None) -> int:
    url = f"{TELEGRAM_API_BASE}{bot_token}/sendMessage"
    params: dict[str, Any] = {
        "chat_id": chat_id,
        "text": text,
        "parse_mode": "HTML",
    }
    if reply_markup:
        params["reply_markup"] = json.dumps(reply_markup)
    data = urllib.parse.urlencode(params).encode("utf-8")
    req = urllib.request.Request(url, data=data, method="POST")
    with urllib.request.urlopen(req, timeout=15) as resp:
        body = json.loads(resp.read())
    if not body.get("ok"):
        raise RuntimeError(f"telegram send failed: {body}")
    return body["result"]["message_id"]
```
